fishのreadコマンド実装

shell   

シェルのreadコマンドは指定したデリミタ（デフォルトは改行）が来るまで文字列を読むので、単純な実装では1byteずつ読む。

一方、fishでは読み込み対象のファイルディスクリプタの特性によって異なり、 ttyやpipeのようなシークできないファイルディスクリプタに対しては1byteずつ、 シークできるものに対してはREAD_CHUNK_SIZEずつ読み込んでいる。

 1static int read_in_chunks(int fd, wcstring &buff, bool split_null) {
 2    int exit_res = STATUS_CMD_OK;
 3    std::string str;
 4    bool eof = false;
 5    bool finished = false;
 6
 7    while (!finished) {
 8        char inbuf[READ_CHUNK_SIZE];
 9        long bytes_read = read_blocked(fd, inbuf, READ_CHUNK_SIZE);
10
11        if (bytes_read <= 0) {
12            eof = true;
13            break;
14        }
15
16        const char *end = std::find(inbuf, inbuf + bytes_read, split_null ? L'\0' : L'\n');
17        long bytes_consumed = end - inbuf;  // must be signed for use in lseek
18        assert(bytes_consumed <= bytes_read);
19        str.append(inbuf, bytes_consumed);
20        if (bytes_consumed < bytes_read) {
21            // We found a splitter. The +1 because we need to treat the splitter as consumed, but
22            // not append it to the string.
23            CHECK(lseek(fd, bytes_consumed - bytes_read + 1, SEEK_CUR) != -1, STATUS_CMD_ERROR)
24            finished = true;
25        } else if (str.size() > read_byte_limit) {
26            exit_res = STATUS_READ_TOO_MUCH;
27            finished = true;
28        }
29    }
30
31    buff = str2wcstring(str);
32    if (buff.empty() && eof) {
33        exit_res = STATUS_CMD_ERROR;
34    }
35
36    return exit_res;
37}

39static int read_one_char_at_a_time(int fd, wcstring &buff, int nchars, bool split_null) {
40    int exit_res = STATUS_CMD_OK;
41    bool eof = false;
42    size_t nbytes = 0;
43
44    while (true) {
45        bool finished = false;
46        wchar_t res = 0;
47        mbstate_t state = {};
48
49        while (!finished) {
50            char b;
51            if (read_blocked(fd, &b, 1) <= 0) {
52                eof = true;
53                break;
54            }
55
56            nbytes++;
57            if (MB_CUR_MAX == 1) {
58                res = (unsigned char)b;
59                finished = true;
60            } else {
61                size_t sz = mbrtowc(&res, &b, 1, &state);
62                if (sz == (size_t)-1) {
63                    memset(&state, 0, sizeof(state));
64                } else if (sz != (size_t)-2) {
65                    finished = true;
66                }
67            }
68        }
69
70        if (nbytes > read_byte_limit) {
71            exit_res = STATUS_READ_TOO_MUCH;
72            break;
73        }
74        if (eof) break;
75        if (!split_null && res == L'\n') break;
76        if (split_null && res == L'\0') break;
77
78        buff.push_back(res);
79        if (nchars > 0 && (size_t)nchars <= buff.size()) {
80            break;
81        }
82    }
83
84    if (buff.empty() && eof) {
85        exit_res = STATUS_CMD_ERROR;
86    }
87
88    return exit_res;
89}

デリミタを超えて読み込んでしまってもlseekを呼び出して読み込みオフセットをデリミタまで戻せばよい。
